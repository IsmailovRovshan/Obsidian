# Работа со свойствами 

```csharp
type.GetProperty("Name")?.SetValue(person, "Иван");
type.GetProperty("LastName")?.SetValue(person, "Петров");
type.GetProperty("Age")?.SetValue(person, 25);

foreach (var prop in type.GetProperties())
{
    Console.WriteLine($"{prop.Name} = {prop.GetValue(person)}");
}
```

```csharp
public class Program
{
    static void Main(string[] args)
    {
        Type personType = typeof(Person);
        object? newPerson = Activator.CreateInstance(personType);

        PropertyInfo nameProperty = personType.GetProperty("Name");
        nameProperty?.SetValue(newPerson, "Tom");

        PropertyInfo ageProperty = personType.GetProperty("Age");
        ageProperty?.SetValue(newPerson, 11);

        if (newPerson is Person person)
        {
            Console.WriteLine(person.Name + " " + person.Age + "");
        }
    }
}
```
